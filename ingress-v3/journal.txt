- creation d'un load balancer externe
    - utilisation du namespace default : 
        - kubectl config set-context --current --namespace=default
    - creation du service
        - kubectl apply -f loadBalancer-service.yaml
        - kubectl apply -f run-bfrk8s-nginx.yaml
        - kubectl expose deployment/bfrk8s-nginx
        - kubectl create configmap nginxconfigmap --from-file=default.conf
        - openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout /tmp/nginx.key -out /tmp/nginx.crt -subj "/CN=my-nginx/O=my-nginx"
        - generer les cles pour les mettres dans le fichier nginxdefault.conf
            - cat /d/tmp/nginx.crt | base64
            - cat /d/tmp/nginx.key | base64
            - copier resultat de ces commandes dans le ficheir nginx-default.conf
        - kubectl apply -f nginxsecrets.yaml

- suppression du loadBalancer
    - kubectl delete svc loadBalancer-service
